import streamlit as st


st.title("Kilometer/Mile Converter")


val = st.number_input("Enter value:", value=0.0)

direction = st.selectbox("Direction:", ["km to mi", "mi to km"])


if direction == "km to mi":

    result = val * 0.621371

    st.success(f"{val} km is {result:.2f} miles")

else:

    result = val / 0.621371

    st.success(f"{val} miles is {result:.2f} km")
